    cin >> n;

    // Kreiranje niza objekata (dinamički)
    // Ovde se N puta poziva Default konstruktor (20x20, "Default")
    Matrica* nizMatrica = new Matrica[n]; 

    // Petlja za izmenu svake matrice (smanjicemo ih na 2x2 radi testa)
    for(int i = 0; i < n; i++) {
        char ime[10];
        sprintf_s(ime, "Mat_%d", i+1); // Pravimo ime Mat_1, Mat_2...
        
        // Koristimo operator dodele da pregazimo default matricu novom manjom

        nizMatrica[i] = Matrica(2, 2, ime); 
        
        cout << "\nUnos za " << ime << ":" << endl;
        nizMatrica[i].unesiElemente();
    }

    cout << "\n--- PRIKAZ SVIH MATRICA ---" << endl;
    for(int i = 0; i < n; i++) {
        nizMatrica[i].prikaziMatricu();
    }

    // Čišćenje niza
    delete[] nizMatrica;

    return 0;
}

void izmeniElement(int red, int kolona, double novaVrednost);

// Funkcija za izmenu pojedinacnog elementa
void Matrica::izmeniElement(int red, int kolona, double novaVrednost) {
    // Provera da li su indeksi validni
    if (red >= 0 && red < brRedova && kolona >= 0 && kolona < brKolona) {
        elementi[red][kolona] = novaVrednost;
        cout << ">> Uspesno izmenjen element [" << red << "][" << kolona << "] na vrednost: " << novaVrednost << endl;
    } else {
        cout << "Greska: Pokusavate da pristupite nepostojecem elementu!" << endl;
    }
}
